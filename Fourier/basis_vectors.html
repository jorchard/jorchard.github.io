DFT Matrix¶

In [33]:
def dftMatrix(N):
    W = np.exp(2.j*np.pi/N)
    M = np.zeros((N,N), dtype=np.csingle)
    for n in range(N):
        for k in range(N):
            M[k,n] = W**(-n*k)
    return M
