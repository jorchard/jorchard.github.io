# RandyA
def RandyA(G, numSteps=10000):
    N = np.shape(G)[0]  # number of nodes
    visits = np.zeros(N)

    j = randint(N)

    for t in range(numSteps):
        visits[j] += 1
        outLinks = find(G[:,j]==1)[0]

        # Test for random teleportation
        if len(outLinks)!=0:
            # Follow an outlink
            j = choice(outLinks)
    return visits

# RandyB
def RandyB(G, numSteps=10000):
    N = np.shape(G)[0]  # number of nodes
    visits = np.zeros(N)

    j = randint(N)
    teleCount = 0  # for counting teleportations
    
    for t in range(numSteps):
        visits[j] += 1
        outLinks = find(G[:,j]==1)[0]
        numOutLinks = len(outLinks)

        # Test for random teleportation
        if numOutLinks!=0:
            # Follow an outlink
            j = outLinks[randint(numOutLinks)]
        else:
            # Teleport
            j = randint(N)
            teleCount += 1
    print('Teleported '+str(teleCount)+' times')    
    return visits

# RandyC
def RandyC(G, numSteps=10000, alpha=0.85):
    N = np.shape(G)[0]  # number of nodes
    visits = np.zeros(N)

    j = randint(N)
    teleCount = 0  # for counting teleportations
    
    for t in range(numSteps):
        visits[j] += 1
        outLinks = find(G[:,j]==1)[0]
        numOutLinks = len(outLinks)

        # Test for random teleportation
        if rand()<alpha:
            # No random teleportation
            if numOutLinks!=0:
                # Follow an outlink
                j = outLinks[randint(numOutLinks)]
            else:
                # Teleport
                j = randint(N)
                teleCount += 1  
        else:
            # Random teleport
            j = randint(N)
            teleCount += 1
    print('Teleported '+str(teleCount)+' times')    
    return visits
